# 引用作为函数重载的条件
```cpp
#include<iostream>
using namespace std;
void print()
{ 
		cout<<"hello world"<<endl;
}
void print(int &a)//引用int &a = 10不合法
{
	cout << "int &a" << endl;
}
void print(const int &a)//引用const int &a = 10合法（编译器创建一个临时变量）
{
	cout << "const int &a" << endl;
}

int main()
{
	int a = 10;
	print(10);
    print(a);
    print();
}
```

# 函数重载碰到默认参数
错误：
```cpp
#include<iostream>
using namespace std;
void print(int a,int b=10)
{ 
		cout<<"print(int a,int b=10)"<<endl;
}
void print(int a)
{
	cout << "print(int a)" << endl;
}


int main()
{
	int a = 10;
	print(a);
    
}
```
函数重载碰到默认参数会出现二义性，报错，尽量避免
